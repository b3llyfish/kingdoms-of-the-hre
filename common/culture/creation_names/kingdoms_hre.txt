swiss_diverged = {
	trigger = {
		OR = {
			capital_county ?= {
				title:k_switzerland = {
					is_de_jure_liege_or_above_target = prev
				}
			}
			has_primary_title = title:k_switzerland
		}
		OR = {
			culture = { has_cultural_pillar = heritage_central_germanic }
			culture = { has_cultural_pillar = heritage_frankish }
			culture = { has_cultural_pillar = heritage_latin }
		}
		any_sub_realm_county = {
			OR = {
				culture = { has_cultural_pillar = heritage_central_germanic }
				culture = { has_cultural_pillar = heritage_frankish }
				culture = { has_cultural_pillar = heritage_latin }
			}
			OR = {
				duchy = title:d_transjurania
				duchy = title:d_currezia
			}
		}
	}
}

slavinian = {
	trigger = {
		OR = {
			capital_county = {
				title:d_nordmark = {
					is_de_jure_liege_or_above_target = prev
				}
			}
			capital_county = {
				title:d_ostmark = {
					is_de_jure_liege_or_above_target = prev
				}
			}
			has_primary_title = title:d_nordmark
			has_primary_title = title:d_ostmark
			has_primary_title = title:k_slavinia
		}
		culture = { has_cultural_pillar = heritage_central_germanic }
		any_sub_realm_county = {
			culture = { has_cultural_pillar = heritage_central_germanic }
			OR = {
				duchy = title:d_nordmark
				duchy = title:d_ostmark
			}
		}
	}
}
