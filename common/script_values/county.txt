required_de_jure_county_percentage_as_decimal = {
	value = 0.5
	if = {
		limit = { root.legitimacy_level = 5 }
		subtract = 0.25
	}
	if = {
		limit = { root.legitimacy_level = 4 }
		subtract = 0.20
	}
	if = {
		limit = { root.legitimacy_level = 3 }
		subtract = 0.15
	}
}
